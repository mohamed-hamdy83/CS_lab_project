cmake_minimum_required(VERSION 3.14)
project(CS_lab_project)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# let cmake handle qt's special macros automatically
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 COMPONENTS Core Gui Widgets Network REQUIRED)

# tell cmake where to look for our header files
include_directories(include)

# ---------------------------------------------------------
# build 1: the main application
# ---------------------------------------------------------
add_executable(ChatApp 
    src/main.cpp 
    src/ChatGUI.cpp 
    src/ChatController.cpp
)
target_link_libraries(ChatApp PRIVATE Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Network)

# ---------------------------------------------------------
# build 2: the unit tests
# ---------------------------------------------------------
enable_testing()

# find googletest 
find_package(GTest REQUIRED)

add_executable(ChatTests 
    tests/ChatControllerTest.cpp 
    src/ChatGUI.cpp 
    src/ChatController.cpp
)
# link qt and googletest to the test program
target_link_libraries(ChatTests PRIVATE Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Network GTest::gtest GTest::gtest_main)
